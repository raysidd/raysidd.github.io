import java.util.Scanner;
class Recipe{
    String Food;
    String Ingredients;
    int Servings;
}
public class MiniLvl3 {
    public static void main(String[] args) {
        String Question = introMessage();
        Message(Question);
        CorrectMessage();
        RecipeContent();
        contMessage();
        EndMessage();

        System.exit(0);
    }

    public static String introMessage() {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Hi there, I'm Chatbot.");
        System.out.println("Try asking me a question about cooking recipes");

        return scanner.nextLine();
    }


    public static void Message(String Question) {
        System.out.println("You asked  " + Question + "," + "correct?");

    }

    public static String CorrectMessage() {
        String Decision;
        Scanner scanner = new Scanner(System.in);
        String text_input = scanner.nextLine();
        Decision = text_input;
        if (Decision.equals("Correct")) {
            System.out.println("Bringing up the current recipes available.");
        } else {
            System.out.println("Sorry, could you please try again.");
        }
        return text_input;
    }

    public static Recipe RecipeInput(String Food, String Ingredients, int Servings) {
        Recipe N = new Recipe();
        N.Food = Food;
        N.Ingredients = Ingredients;
        N.Servings = Servings;
        return N;
    }

    public static void RecipeContent() {
        Recipe Pizza = RecipeInput("Pizza", "1kg Pizza Dough,500g Tomato Paste & 500g Mozzarella Cheese", 8);
        Recipe FriedChicken = RecipeInput("Fried Chicken", "850g Chicken, 125g Chicken Fry Mix", 10);
        Recipe FriedRice = RecipeInput("Fried Rice", "1kg Rice, 500g Mixed Vegetables, 2 tbsp Salt", 12);
        Scanner scanner = new Scanner(System.in);
        System.out.println("We have the recipes of the following foods:");
        System.out.println(Pizza.Food);
        System.out.println(FriedChicken.Food);
        System.out.println(FriedRice.Food);
        System.out.println("Which of the following dishes would you like to know about");
        String text_input = scanner.nextLine();

        if (text_input.equals("Pizza")) {
            System.out.println("The ingredients required are: " + Pizza.Ingredients + " and this dish serves" + Pizza.Servings + " people.");
        } else if (text_input.equals("Fried Chicken")) {
            System.out.println("The ingredients required are: " + FriedChicken.Ingredients + " and this dish serves " + FriedChicken.Servings + " people.");
        } else if (text_input.equals("Fried Rice")) {
            System.out.println("The ingredients required are: " + FriedRice.Ingredients + " and this dish serves " + FriedRice.Servings + " people.");
        } else System.out.println("Sorry we don't have a recipe for that :(");
    }

    public static void contMessage() {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Would you like to know about the other two dishes?");
        String text_input = scanner.nextLine();

        while (true) {
            if (text_input.equals("Yes")) ;

            RecipeContent();
            RecipeContent();
            break;
        }


    }

    public static void EndMessage() {
        Scanner scanner = new Scanner(System.in);
        String text_input = scanner.nextLine();
        if (text_input.equals("No")) {
            System.out.println("Goodbye, have a great day!");
            System.exit(0);

        }
    }
}
